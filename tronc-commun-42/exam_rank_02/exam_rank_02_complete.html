1. Squelette main()

#include <unistd.h>

int	main(int argc, char **argv)
{
	if (argc == 2)
		my_function(argv[1]);
	write(1, "\n", 1);

}
Le \n est TOUJOURS après le if, pas dedans. Comme ça il s'affiche même si argc est faux.

2. ft_putchar + ft_putnbr

{
	write(1, &c, 1);

void	ft_putnbr(int n)
{
	char	c;

	if (n < 0)
	{
		ft_putchar('-');
		n = -n;

		ft_putnbr(n / 10);

}

6. ft_atoi

int	ft_atoi(const char *str)
{
	int	i;
	int	sign;
	int	result;

	i = 0;
	sign = 1;
	result = 0;
	while (str[i] == ' ' || (str[i] >= 9 && str[i] <= 13))
		i++;
	if (str[i] == '-' || str[i] == '+')
	{
		if (str[i] == '-')
			sign = -1;
		i++;
	}
	while (str[i] >= '0' && str[i] <= '9')
	{
		result = result * 10 + (str[i] - '0');
		i++;
	}
	return (sign * result);
}

// Énoncé

Assignment name  : repeat_alpha
Expected files   : repeat_alpha.c
Allowed functions: write

Write a program that takes a string and displays it repeating each alphabetical character as many times as its alphabetical index, followed by a newline.
'a' becomes 'a', 'b' becomes 'bb', 'e' becomes 'eeeee', etc...
Case remains unchanged. If the number of arguments is not 1, just display a newline.

$>./repeat_alpha "abc"
abbccc
$>./repeat_alpha "Alex." | cat -e
Alllllllllllleeeeexxxxxxxxxxxxxxxxxxxxxxxx.$
'a' = 1 fois, 'b' = 2 fois, ... 'z' = 26 fois. Index = c - 'a' + 1.
#include <unistd.h>

void	repeat_alpha(char *str)
{
	int	i;
	int	rep;

	i = 0;
	while (str[i])
	{
		rep = 1;
		if (str[i] >= 'a' && str[i] <= 'z')
			rep = str[i] - 'a' + 1;
		else if (str[i] >= 'A' && str[i] <= 'Z')
			rep = str[i] - 'A' + 1;
		while (rep-- > 0)
			write(1, &str[i], 1);
		i++;
	}
}

int	main(int argc, char **argv)
{
	if (argc == 2)
		repeat_alpha(argv[1]);
	write(1, "\n", 1);
	return (0);
}

// ft_strcspn — longueur du prefixe sans chars de reject
size_t	ft_strcspn(const char *s, const char *reject)
{
	size_t	i;
	size_t	j;

	i = 0;
	while (s[i])
	{
		j = 0;
		while (reject[j])
		{
			if (s[i] == reject[j])
				return (i);
			j++;
		}
		i++;
	}
	return (i);
}

// ft_strspn — longueur du prefixe avec SEULEMENT chars de accept
size_t	ft_strspn(const char *s, const char *accept)
{
	size_t	i;
	size_t	j;

	i = 0;
	while (s[i])
	{
		j = 0;
		while (accept[j] && accept[j] != s[i])
			j++;
		if (!accept[j])
			return (i);
		i++;
	}
	return (i);
}

// ft_strpbrk — pointeur vers 1er char de s1 trouvé dans s2
char	*ft_strpbrk(const char *s1, const char *s2)
{
	int	i;
	int	j;

	i = 0;
	while (s1[i])
	{
		j = 0;
		while (s2[j])
		{
			if (s1[i] == s2[j])
				return ((char *)&s1[i]);
			j++;
		}
		i++;
	}
	return (NULL);
}

		word[i] = str[start];
		i++;
		start++;
	}
	word[i] = '\0';
	return (word);
}

char	**ft_split(char *str)
{
	char	**res;
	int		i;
	int		w;
	int		start;

	res = malloc(sizeof(char *) * (count_words(str) + 1));
	if (!res)
		return (NULL);
	i = 0;
	w = 0;
	while (str[i])
	{
		while (str[i] && is_sep(str[i]))
			i++;
		if (str[i])
		{
			start = i;
			while (str[i] && !is_sep(str[i]))
				i++;
			res[w++] = get_word(str, start, i);
		}
	}
	res[w] = NULL;
	return (res);
}
LE boss de l'exam. Apprends le pattern skip_sep/skip_word par cœur.

ASCII

'0' = 48    '9' = 57

'a' = 97    'z' = 122
' ' = 32    '\t' = 9    '\n' = 10
Diff maj/min = 32

Formules

Strlen         : while(s[len]) len++;
Rotation N     : c = (c - 'a' + N) % 26 + 'a'
Mirror         : c = 'z' - (c - 'a')
To upper       : c -= 32
To lower       : c += 32
Is alpha       : (c>='a'&&c<='z')||(c>='A'&&c<='Z')
Power of 2     : n && !(n & (n-1))
GCD (Euclide)  : while(b){t=b;b=a%b;a=t;}
LCM            : a / gcd(a,b) * b
Facteurs       : while(n%d==0){print;n/=d;} d++;

cc -Wall -Wextra -Werror fichier.c -o fichier
./fichier "arg1" "arg2" | cat -e
